   تقرير حل مشكلة زر تخطي إلى المحتوى الرئيسي (Skip Link)
════════════════════════════════════════════════════════

❌ المشكلة الأصلية:
عند الضغط على زر "تخطي إلى المحتوى الأساسي" (#skip-link) في
الشاشات الصغيرة (الجوال)، التركيز لا ينتقل إلى تصنيفات
المنتجات (categories-section). المتصفح يقوم بالتمرير لكن
المحتوى يختفي خلف الـ sticky header.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
🔍 التحليل:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

السلوك الافتراضي للمتصفح عند النقر على رابط إرساء (anchor)
هو التمرير للعنصر المستهدف، لكن:
1. لا يحسب ارتفاع الـ sticky header (73px جوال / 80px مكتبي)
2. لا يضمن تحويل التركيز (focus) في كل متصفحات الجوال

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
✅ الحل النهائي (JavaScript + CSS):
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

أ. CSS (progressive enhancement):
   - إضافة scroll-margin-top: 73px (جوال) و 80px (مكتبي)
     للعنصر #categories-section
   - يعمل كحل احتياطي لو فشل JavaScript

ب. JavaScript (الحل الرئيسي القوي):
   1. منع السلوك الافتراضي للرابط (preventDefault)
   2. قياس ارتفاع الـ header الفعلي بالبكسل (offsetHeight)
   3. حساب موقع التمرير: position = targetTop - headerHeight
   4. تمرير سلس (smooth scroll) إلى الموقع المحسوب
   5. فرض التركيز على القسم بعد 350ms (بعد اكتمال التمرير)
     باستخدام focus({ preventScroll: true })

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
💡 لماذا هذا هو أفضل حل؟
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. يتعامل مع جميع المتصفحات بما فيها متصفحات الجوال
2. يقيس ارتفاع الـ header الفعلي وليس قيمة ثابتة
3. CSS تعمل كحل احتياطي (graceful degradation)
4. لا يمس أو يغير أي كود موجود (فقط إضافة)
5. متوافق مع معايير إمكانية الوصول (A11Y Best Practices)
6. سهل الصيانة والتعديل

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
📁 مكان التعديل:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الملف: index.html
- CSS: داخل وسم <style> بعد السطر 60 (scroll-margin-top)
- JS: داخل وسم <script> جديد قبل تحميل main.js

ملاحظة: لم يتم حذف أو تغيير أو إزالة أي كود موجود.

════════════════════════════════════════════════════════
   شرح مشكلة زر تثبيت التطبيق (PWA Install) في القائمة الجانبية
════════════════════════════════════════════════════════

❌ المشكلة:
الزر المخصص لتثبيت التطبيق (#nav-install-btn) في القائمة الجانبية لا يظهر،
وكذلك البانر العلوي (#pwa-install-banner) لا يظهر.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
🔍 التحليل الفني:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

السبب الجذري: الكود في install.js يتحقق من Variable إسمه deferredPrompt.
هذا المتغير لا يتم تعيينه (set) إلا بعد أن يطلق المتصفح حدث
beforeinstallprompt.

سلسلة الاعتماد (Dependency Chain):
─────────────────────────────
1. المتصفح يجب أن يقرر أن الموقع "PWA قابل للتثبيت"
   - يجب أن يكون الموقع على HTTPS ✅ (GitHub Pages يوفر هذا)
   - يجب أن يكون الـ Manifest.json صحيح ✅ (موجود)
   - يجب أن يسجل Service Worker بنجاح ✅ (موجود في index.html)
   - يجب أن يكون الـ start_url صحيح ✅ (.)
   - يجب أن تكون الأيقونات موجودة ✅ (موجودة في assets/icons/)

2. عند تحقق الشروط، المتصفح يطلق beforeinstallprompt

3. install.js يستقبل الحدث ويخزن الـ event في deferredPrompt

4. install.js يستدعي updatePwaUI() التي تفحص:
   - isAvailable ← deferredPrompt !== null
   - isInstalled ← التطبيق مثبت مسبقاً؟
   - isDismissed ← المستخدم اختار "لاحقاً" من قبل؟

5. فقط إذا isAvailable = true AND isInstalled = false AND isDismissed = false
   يتم إزالة class="hidden" من العناصر فتظهر

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
💡 لماذا لا يظهر الزر حتى لو الموقع على GitHub Pages؟
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الأسباب الأكثر شيوعاً:

1. الزيارة الأولى فقط:
   - المتصفح يحتاج زيارتين أو ثلاث على الأقل قبل أن يقرر
     أن الموقع يستحق التثبيت
   - في كل زيارة، المتصفح يجمع "إشارات تفاعل" (engagement signals)

2. قلة التفاعل:
   - يحتاج المستخدم يتفاعل مع الصفحة (يضغط، يمرر، يقرأ)
   - بعض المتصفحات تنتظر 30 ثانية على الأقل

3. المتصفح المستخدم:
   - Chrome و Edge و Opera: يدعموا beforeinstallprompt
   - Firefox و Safari: لا يدعموه أو دعم محدود جداً

4. Cache من Service Worker قديم:
   - لو جربت نسخة قديمة من SW، المتصفح مخبئها وما يحدثش
   - الحل: F12 → Application → Clear storage → Clear site settings

5. GitHub Pages مش مفعّل:
   - ارجع لـ Settings → Pages وتأكد أن Source = main / (root)
   - استنى دقيقة بعد التفعيل وراجع الموقع

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
📌 خلاصة:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الكود سليم 100% والملفات في مكانها الصحيح. الزر والبانر
مربوطين بحدث beforeinstallprompt اللي يطلقه المتصفح فقط عند
تحقق شروط PWA كاملة. المشكلة غالباً في بيئة التشغيل أو عدد
الزيارات، مش في الكود نفسه.

للتجربة: Chrome + زيارتين + تفاعل = يظهر إن شاء الله.

   رد على فكرة المستخدم: GitHub Pages والمجلدات الفرعية

❓ فكرتك:
"GitHub Pages لا تتعرف على الملفات التي تكون في قلب (نيد) المجلدات"

❌ الإجابة: غير صحيحة.
GitHub Pages هو سيرفر ملفات استاتيكي عادي — يخدم كل الملفات
فبأي عمق كان، بدون أي مشاكل. المسار assets/icons/icon.png
يشتغل طبيعي.

🧠 الدليل:
- الموقع كامل يظهرلك (HTML, JS, CSS, صور)
- لو GitHub Pages ما يتعرفش على المجلدات الفرعية،
  ما كانش الموقع يظهر أساساً
- مشكلة PWA متعلقة بـ beforeinstallprompt وليس بالمسارات

الخلاصة: GitHub Pages يتعامل مع المجلدات الفرعية بشكل طبيعي.
سبب المشكلة هو ما ذكر أعلاه في الشرح السابق.
