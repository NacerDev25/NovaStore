   تقرير حل مشكلة زر تخطي إلى المحتوى الرئيسي (Skip Link)
════════════════════════════════════════════════════════

❌ المشكلة الأصلية:
عند الضغط على زر "تخطي إلى المحتوى الأساسي" (#skip-link) في
الشاشات الصغيرة (الجوال)، التركيز لا ينتقل إلى تصنيفات
المنتجات (categories-section). المتصفح يقوم بالتمرير لكن
المحتوى يختفي خلف الـ sticky header.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
🔍 التحليل:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

السلوك الافتراضي للمتصفح عند النقر على رابط إرساء (anchor)
هو التمرير للعنصر المستهدف، لكن:
1. لا يحسب ارتفاع الـ sticky header (73px جوال / 80px مكتبي)
2. لا يضمن تحويل التركيز (focus) في كل متصفحات الجوال

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
✅ الحل النهائي (JavaScript + CSS):
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

أ. CSS (progressive enhancement):
   - إضافة scroll-margin-top: 73px (جوال) و 80px (مكتبي)
     للعنصر #categories-section
   - يعمل كحل احتياطي لو فشل JavaScript

ب. JavaScript (الحل الرئيسي القوي):
   1. منع السلوك الافتراضي للرابط (preventDefault)
   2. قياس ارتفاع الـ header الفعلي بالبكسل (offsetHeight)
   3. حساب موقع التمرير: position = targetTop - headerHeight
   4. تمرير سلس (smooth scroll) إلى الموقع المحسوب
   5. فرض التركيز على القسم بعد 350ms (بعد اكتمال التمرير)
     باستخدام focus({ preventScroll: true })

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
💡 لماذا هذا هو أفضل حل؟
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. يتعامل مع جميع المتصفحات بما فيها متصفحات الجوال
2. يقيس ارتفاع الـ header الفعلي وليس قيمة ثابتة
3. CSS تعمل كحل احتياطي (graceful degradation)
4. لا يمس أو يغير أي كود موجود (فقط إضافة)
5. متوافق مع معايير إمكانية الوصول (A11Y Best Practices)
6. سهل الصيانة والتعديل

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
📁 مكان التعديل:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الملف: index.html
- CSS: داخل وسم <style> بعد السطر 60 (scroll-margin-top)
- JS: داخل وسم <script> جديد قبل تحميل main.js

ملاحظة: لم يتم حذف أو تغيير أو إزالة أي كود موجود.
